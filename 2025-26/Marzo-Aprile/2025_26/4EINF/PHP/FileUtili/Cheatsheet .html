🔌 Connessione al Database

Da fare SEMPRE come primo passo — senza connessione non funziona nulla.

// mysqli_connect(host, utente, password, database)
$con = mysqli_connect("localhost", "root", "", "nome_database");

// Controllo che la connessione sia riuscita
if ($con->connect_errno) {
    die("Connessione fallita: " . $con->connect_errno
        . " - " . $con->connect_error);
}

// ... tutta la logica della pagina ...

// Alla fine: chiudo sempre la connessione
$con->close();
⚠ Con XAMPP: host="localhost", utente="root", password="" (vuota)

📊 Query SELECT — Leggere dati

// ── SELECT base: prende tutte le righe ──
$sql = "SELECT nome, cognome, email FROM utenti";
$rs  = $con->query($sql);

// Controllo esito query
if (!$rs) {
    die("Query fallita: " . $con->errno . " - " . $con->error);
}

// Leggo riga per riga con fetch_assoc()
// $row è un array: $row["nome"], $row["email"], ecc.
while ($row = $rs->fetch_assoc()) {
    echo $row["nome"] . " " . $row["cognome"] . "<br>";
}

// ── SELECT con WHERE: filtra per condizione ──
$sql = "SELECT * FROM utenti WHERE id = $id";

// ── SELECT con ORDER BY: ordina risultati ──
$sql = "SELECT * FROM prodotti ORDER BY prezzo DESC";
//  ASC = crescente (default) | DESC = decrescente

// ── SELECT singola riga (senza while) ──
$rs  = $con->query("SELECT * FROM utenti WHERE id = $id");
$row = $rs->fetch_assoc();  // una sola riga → no while
echo $row["nome"];

🔗 JOIN — Unire due tabelle

Usa JOIN quando i dati che ti servono sono divisi su due tabelle legate da una chiave esterna.
// Esempio: utenti (id_abbonamento) → abbonamenti (id)
// JOIN "unisce" le righe in base alla colonna in comune

$sql = "SELECT u.nome, u.cognome, u.email,
                a.tipo, a.costo_mensile
        FROM utenti u
        JOIN abbonamenti a ON u.id_abbonamento = a.id";

// u. e a. sono alias delle tabelle: evitano ambiguità
// se entrambe le tabelle hanno una colonna "id"

// ── JOIN con WHERE ──
$sql = "SELECT u.*, a.tipo, a.costo_mensile
        FROM utenti u
        JOIN abbonamenti a ON u.id_abbonamento = a.id
        WHERE u.id = $id_utente";

// ── JOIN con rinomina colonne (AS) ──
// Utile quando due tabelle hanno colonne con lo stesso nome
$sql = "SELECT
            u.id AS id_utente,
            u.nome, u.cognome,
            a.id AS id_abbonamento,
            a.tipo
        FROM utenti u
        JOIN abbonamenti a ON u.id_abbonamento = a.id";

📈 GROUP BY, COUNT, HAVING

Usato per contare/raggruppare righe per categoria.

// COUNT(*): conta le righe per gruppo
// GROUP BY: raggruppa per una colonna
$sql = "SELECT tipo, COUNT(*) AS totale
        FROM abbonamenti a
        JOIN utenti u ON a.id = u.id_abbonamento
        GROUP BY tipo";
// $row["totale"] = numero utenti per tipo piano

// HAVING: come WHERE ma si applica DOPO il GROUP BY
$sql = "SELECT citta, COUNT(*) AS tot
        FROM utenti
        GROUP BY citta
        HAVING tot > 2";
// Restituisce solo le città con più di 2 utenti

// MAX, MIN, SUM, AVG
$sql = "SELECT MAX(costo_mensile), MIN(costo_mensile),
               AVG(costo_mensile)
        FROM abbonamenti";

🏗 SQL — CREATE TABLE

CREATE DATABASE IF NOT EXISTS nome_db
    CHARACTER SET utf8mb4;

USE nome_db;

DROP TABLE IF EXISTS tabella_figlia;  -- elimina prima la figlia
DROP TABLE IF EXISTS tabella_padre;

CREATE TABLE prodotti (
    id          INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    nome        VARCHAR(100) NOT NULL,
    prezzo      DECIMAL(8,2)  NOT NULL,
    descrizione TEXT,
    attivo      BOOLEAN NOT NULL DEFAULT TRUE,
    creato_il   DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- Tabella FIGLIA con chiave esterna (FK)
CREATE TABLE ordini (
    id          INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    id_prodotto INT UNSIGNED NOT NULL,
    quantita    INT UNSIGNED NOT NULL DEFAULT 1,
    CONSTRAINT fk_ordini_prodotti
        FOREIGN KEY (id_prodotto) REFERENCES prodotti(id)
        ON UPDATE CASCADE ON DELETE RESTRICT
) ENGINE=InnoDB;
Tipi utili: VARCHAR(n) testo breve · TEXT testo lungo · INT intero · DECIMAL(m,d) decimale · BOOLEAN vero/falso · DATE data · DATETIME data+ora · ENUM('a','b') valori fissi

➕ SQL — INSERT

-- Inserimento singola riga
INSERT INTO prodotti (nome, prezzo, descrizione)
VALUES ('Maglietta', 19.99, 'Cotone 100%');

-- Inserimento multiplo (più righe in una query)
INSERT INTO prodotti (nome, prezzo)
VALUES
    ('Cappello', 9.99),
    ('Cintura',  14.99),
    ('Borsa',    39.99);

✏️ SQL — UPDATE e DELETE

-- UPDATE: modifica righe esistenti
UPDATE prodotti
SET   prezzo = 24.99, nome = 'Maglietta Premium'
WHERE id = 1;

-- DELETE: elimina righe
DELETE FROM prodotti WHERE id = 5;

-- ⚠ ATTENZIONE: senza WHERE aggiorna/elimina TUTTO!
⚠ Metti SEMPRE la WHERE in UPDATE e DELETE!
// Eseguire UPDATE/DELETE in PHP (no fetch_assoc, usano query diretta)
$sql = "UPDATE prodotti SET prezzo = 24.99 WHERE id = $id";
$ok  = $con->query($sql);
if (!$ok) { die("Errore: " . $con->error); }
echo "Righe modificate: " . $con->affected_rows;

📝 Form HTML — GET e POST

GET I dati appaiono nell'URL (?nome=Mario) – usa per ricerche/filtri
POST I dati non appaiono nell'URL – usa per invio dati sensibili/azioni
<!-- FORM POST sulla stessa pagina (action="") -->
<form action="" method="POST">
    <input type="text" name="username" placeholder="Nome">
    <select name="tipo">
        <option value="base">Base</option>
        <option value="premium">Premium</option>
    </select>
    <!-- input hidden: non visibile, inviato col form -->
    <input type="hidden" name="azione" value="cerca">
    <button type="submit" name="invia" value="1">Invia</button>
</form>

<!-- FORM GET: passa parametri nell'URL -->
<form action="dettaglio.php" method="GET">
    <input type="hidden" name="id" value="5">
    <button type="submit">Vai al dettaglio</button>
</form>

<!-- Link con parametro GET nell'URL (senza form) -->
<a href="dettaglio.php?id=<?= $utente["id"] ?>">Dettaglio</a>

🔎 isset() — Controllare GET e POST

isset($var) restituisce TRUE se la variabile esiste e non è NULL. Usalo SEMPRE prima di leggere $_GET o $_POST.
// ── Modalità GET/POST: schema tipico della verifica ──
if (!isset($_POST["abbonati"])) {
    // Nessun POST → mostra home page
} else {
    // POST ricevuto → mostra lista / elabora dati
}

// ── Lettura sicura da GET ──
if (isset($_GET["id"])) {
    $id = intval($_GET["id"]);   // converto sempre in intero!
}

// ── Lettura sicura da POST ──
if (isset($_POST["username"])) {
    $nome = htmlspecialchars($_POST["username"]); // sanificazione XSS
}

// ── Valore di default se il parametro non c'è ──
$pagina = isset($_GET["p"]) ? intval($_GET["p"]) : 1;

// ── Controllare se un pulsante submit è stato premuto ──
if (isset($_POST["cerca"])) { // cerca = name del button
    // il pulsante "cerca" è stato premuto
}

📦 Array e foreach

// ── Creare e riempire array ──
$frutti = [];                  // array vuoto
$frutti[] = "mela";           // aggiunge in fondo
$frutti[] = "banana";
count($frutti);               // → 2 (numero elementi)

// ── Array associativo ──
$utente = ["nome" => "Mario", "eta" => 30];
echo $utente["nome"];          // → Mario

// ── foreach su array semplice ──
foreach ($frutti as $frutto) {
    echo $frutto . "<br>";
}

// ── foreach su array di array (tipico da DB) ──
foreach ($premium as $utente) {
    echo $utente["nome"] . " - " . $utente["email"] . "<br>";
}

// ── Raccogliere righe DB in array (schema tipico verifica) ──
$premium  = [];
$standard = [];
$base     = [];

while ($row = $rs->fetch_assoc()) {
    if     ($row["tipo"] == "premium")  $premium[]  = $row;
    elseif ($row["tipo"] == "standard") $standard[] = $row;
    else                                  $base[]     = $row;
}

⚡ Operatore Ternario e ucfirst/number_format

Il ternario è un if/else in una riga: condizione ? se_vero : se_falso
// if/else classico:
if ($attivo == 1) { $testo = "Attivo";   }
else             { $testo = "Inattivo"; }

// Stessa cosa con l'operatore ternario:
$testo = ($attivo == 1) ? "Attivo" : "Inattivo";

// Ternario direttamente nell'echo:
echo $row["download_offline"] ? "✅ Sì" : "❌ No";

// ── Funzioni di formattazione utili ──
ucfirst("premium");            // → "Premium" (prima lettera maiuscola)
strtoupper("ciao");           // → "CIAO"
strtolower("CIAO");           // → "ciao"
number_format(19.9, 2);       // → "19.90" (2 decimali)
date("d/m/Y");                 // → data odierna es. "02/05/2026"
strlen("ciao");               // → 4 (lunghezza stringa)

🛡 Sicurezza base — intval e htmlspecialchars

⚠ In verifica usa sempre questi due quando leggi dati dall'utente!
// intval(): converte in intero → protegge da SQL Injection base
// "5; DROP TABLE utenti" → intval() → 5 (il resto viene ignorato)
$id = intval($_GET["id"]);

// htmlspecialchars(): converte caratteri HTML speciali in entità
// Evita XSS (Cross-Site Scripting)
// <script>alert('hack')</script> → diventa testo innocuo
$nome = htmlspecialchars($_POST["nome"]);

// Uso tipico: id da URL usato in query
$id  = intval($_GET["id"]);
$sql = "SELECT * FROM utenti WHERE id = $id";

🔀 Redirect — header()

header() deve essere chiamata PRIMA di qualsiasi output HTML, altrimenti dà errore.
// Reindirizza a un'altra pagina
// ⚠ DEVE essere prima di qualsiasi echo o HTML
header("Location: index.php");
exit;   // ferma lo script dopo il redirect — OBBLIGATORIO!

// Redirect con parametro GET:
header("Location: dettaglio.php?id=" . $id);
exit;

// Schema tipico: se manca il parametro → torna alla home
if (!isset($_GET["id"])) {
    header("Location: index.php");
    exit;
}

✅ Booleani dal DB — come gestirli

MySQL salva i BOOLEAN come 1 (TRUE) o 0 (FALSE). PHP li riceve come stringa "1" o "0".
// ── Metodo 1: confronto diretto ──
if ($row["attivo"] == 1) {
    echo '<span class="badge bg-success">Attivo</span>';
} else {
    echo '<span class="badge bg-danger">Inattivo</span>';
}

// ── Metodo 2: operatore ternario (più compatto) ──
echo $row["download_offline"] ? "✅ Sì" : "❌ No";
echo $row["pubblicita"]       ? "❌ Presente" : "✅ Assente";

// ── In SQL: filtrare per booleano ──
$sql = "SELECT * FROM utenti WHERE attivo = TRUE";
$sql = "SELECT * FROM prodotti WHERE disponibile = 1";

📄 index.php

<?php
if (!isset($_POST["mostra"])) {
    // ── HOME PAGE ──
    echo '<form action="" method="POST">
        <button name="mostra" value="1">
            Mostra dati
        </button>
    </form>';
} else {
    // ── LISTA DATI ──
    $con = mysqli_connect(...);
    if ($con->connect_errno)
        die($con->connect_error);

    $sql = "SELECT ... FROM ... JOIN ...";
    $rs  = $con->query($sql);

    $gruppoA = [];
    $gruppoB = [];
    while ($row = $rs->fetch_assoc()) {
        if ($row["tipo"] == "A") $gruppoA[] = $row;
        else                   $gruppoB[] = $row;
    }

    echo '<div class="row">';
    echo '<div class="col-6">';
    foreach ($gruppoA as $u) {
        echo '<a href="dettaglio.php?id='
             . $u["id"] . '">';
        echo '<div class="card mb-2 p-2">';
        echo $u["nome"];
        echo '</div></a>';
    }
    echo '</div></div>';
    echo '<a href="index.php">← Home</a>';
    $con->close();
}
?>
